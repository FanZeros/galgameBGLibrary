// --- Rate Limiter (1 request per 60 seconds per IP, in-memory) ---
const RATE_LIMIT_WINDOW_MS = 60 * 1000; // 1 minute
const rateLimitMap = new Map(); // IP -> { count, resetTime }

// Auto-clean expired entries every 5 minutes
setInterval(() => {
    const now = Date.now();
    for (const [ip, entry] of rateLimitMap) {
        if (now > entry.resetTime) rateLimitMap.delete(ip);
    }
}, 5 * 60 * 1000);

const rateLimiter = (req, res, next) => {
    const ip = req.ip;
    const now = Date.now();
    const entry = rateLimitMap.get(ip);

    if (entry && now < entry.resetTime) {
        const retryAfter = Math.ceil((entry.resetTime - now) / 1000);
        console.log(`⛔ Rate limited: ${ip} (retry after ${retryAfter}s)`);
        res.set('Retry-After', String(retryAfter));
        return res.status(429).json({
            error: `请求过于频繁，请在 ${retryAfter} 秒后重试。`,
            retryAfterMs: entry.resetTime - now
        });
    }

    rateLimitMap.set(ip, { count: 1, resetTime: now + RATE_LIMIT_WINDOW_MS });
    next();
};

// --- Input Validation Middleware ---
const MAX_PLAYER_INPUT_LENGTH = 500;

const validateGenerateInput = (req, res, next) => {
    const body = req.body;

    // Check required fields exist and have correct types
    if (!body || typeof body !== 'object') {
        return res.status(400).json({ error: 'Request body must be a JSON object.' });
    }

    const { playerInput, currentStoryContext, characters, existingBackgrounds, existingBgms } = body;

    // playerInput: required, must be string, max length, strip HTML
    if (typeof playerInput !== 'string' || playerInput.trim().length === 0) {
        return res.status(400).json({ error: 'playerInput must be a non-empty string.' });
    }
    // Sanitize: strip HTML tags and limit length
    body.playerInput = playerInput.replace(/<[^>]*>/g, '').trim().slice(0, MAX_PLAYER_INPUT_LENGTH);

    // currentStoryContext: required object with id
    if (!currentStoryContext || typeof currentStoryContext !== 'object' || !currentStoryContext.id) {
        return res.status(400).json({ error: 'currentStoryContext must be an object with an id.' });
    }

    // characters: must be array if present
    if (characters !== undefined && !Array.isArray(characters)) {
        return res.status(400).json({ error: 'characters must be an array.' });
    }

    // existingBackgrounds: must be array if present
    if (existingBackgrounds !== undefined && !Array.isArray(existingBackgrounds)) {
        return res.status(400).json({ error: 'existingBackgrounds must be an array.' });
    }

    // existingBgms: must be array if present
    if (existingBgms !== undefined && !Array.isArray(existingBgms)) {
        return res.status(400).json({ error: 'existingBgms must be an array.' });
    }

    // aiGuidance: optional string, sanitize if present
    if (body.aiGuidance && typeof body.aiGuidance === 'string') {
        body.aiGuidance = body.aiGuidance.replace(/<[^>]*>/g, '').trim().slice(0, 1000);
    }

    next();
};

const extractJSON = (str) => {
    let firstOpen = str.indexOf('{');
    let firstArray = str.indexOf('[');

    // Determine if it's an object or array (whichever starts first)
    if (firstOpen === -1 && firstArray === -1) return null;

    let start, endChar;

    // If array starts before object (or object doesn't exist), assume array
    if (firstArray !== -1 && (firstOpen === -1 || firstArray < firstOpen)) {
        start = firstArray;
        endChar = ']'; // Match the correct closing bracket type
    } else {
        start = firstOpen;
        endChar = '}'; // Match the correct closing bracket type
    }

    const end = str.lastIndexOf(endChar);

    if (start !== -1 && end !== -1 && end > start) {
        return str.substring(start, end + 1);
    }
    return null;
};

const cleanJsonString = (str) => {
    if (!str) return "{}";
    let cleaned = str.trim();

    // 1. Remove markdown code blocks (keep content)
    cleaned = cleaned.replace(/^```json/gm, '').replace(/^```/gm, '').replace(/```$/gm, '');

    // 2. Try to extract valid JSON substring (Iterative approach for robustness)
    let cursor = 0;
    while (cursor < cleaned.length) {
        const substr = cleaned.substring(cursor);
        const extracted = extractJSON(substr);

        if (!extracted) {
            break;
        }

        // Validate if this extraction is parseable
        try {
            JSON.parse(extracted);
            return extracted; // Found valid JSON!
        } catch (e) {
            // If invalid, move cursor past the first '{' of this attempt and try searching again
            // This handles cases like: "**Thinking** { broken } ... { valid }"
            const firstBrace = substr.indexOf('{');
            if (firstBrace !== -1) {
                cursor += firstBrace + 1;
            } else {
                break;
            }
        }
    }

    // Fallback: Just return the first brace extraction and hope for the best
    const fallback = extractJSON(cleaned);
    if (fallback) return fallback;

    throw new Error(`No JSON object found. Content snippet: ${cleaned.substring(0, 100)}...`);
};

// --- Semantic Check Helper ---
const checkSemanticMatch = async (input, historyEntries, context, characterNames, characterState, apiKey) => {
    if (!historyEntries || historyEntries.length === 0) return -1;

    // Fast path: exact string match (avoids unnecessary AI API call)
    const exactIdx = historyEntries.findIndex(h => h.input === input);
    if (exactIdx !== -1) {
        console.log(`⚡ Exact match found at index ${exactIdx}: "${historyEntries[exactIdx].input}"`);
        return exactIdx;
    }

    console.log(`🔍 Checking semantic match for: "${input}" against ${historyEntries.length} options...`);

    const prompt = `
    You are a semantic matching engine for a text adventure game.
    
    **CURRENT STORY CONTEXT:**
    ${JSON.stringify(context)}

    **KNOWN CHARACTERS:**
    ${JSON.stringify(characterNames)}

    **CURRENT CHARACTER STATE:**
    ${JSON.stringify(characterState || {})}

    **PLAYER INPUT:** "${input}"
    
    **HISTORY CANDIDATES (Input + State):**
    ${historyEntries.map((h, i) => `${i}: Input="${h.input}" | State=${JSON.stringify(h.state || {})}`).join('\n')}
    
    Task: 
    Determine if the PLAYER INPUT conveys the essentially SAME intention as any of the EXISTING CHOICES considering the story context AND Character State.
    
    **CRITICAL RULES:**
    1. **Semantic Match**: The action/intention must be effectively the same.
    2. **State Consistency (VERY IMPORTANT)**: 
       - Even if the input is identical, check if the *Character State* (affection, mood) is compatible.
       - A high-affection reaction (e.g. "I kiss him") might be stored. If the Current State is now low affection, this is a MISMATCH (return -1).
       - Minor state differences (e.g. affection 50 vs 55) are accceptable matches. Large shifts (Happy vs Angry, High vs Low) are mismatches.
    
    - If it matches, return the index of the best match.
    - If the input is a distinct new action OR the state is too different, return -1.
    
    **OUTPUT JSON ONLY:** { "matchIndex": number }
    `;

    try {
        const response = await fetch('https://api.vectorengine.ai/v1beta/models/gemini-2.5-flash:generateContent', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json', 'x-goog-api-key': apiKey },
            body: JSON.stringify({
                contents: [{ parts: [{ text: prompt }] }],
                generationConfig: { responseMimeType: "application/json" }
            })
        });

        if (!response.ok) {
            console.warn("Semantic check API failed, skipping cache check.");
            return -1;
        }

        const data = await response.json();
        const text = data.candidates?.[0]?.content?.parts?.[0]?.text;

        if (text) {
            const result = JSON.parse(cleanJsonString(text));
            const idx = typeof result.matchIndex === 'number' ? result.matchIndex : -1;
            if (idx !== -1) console.log(`✅ Match found! Index ${idx}: "${historyEntries[idx].input}"`);
            else console.log(`⚪ No match found (New Action or State Mismatch). Generating new branch.`);
            return idx;
        }
    } catch (e) {
        console.error("Semantic check failed:", e);
    }
    return -1;
};

        if (GAME_SERVER_TOKEN) {
            const expectedToken = `Bearer ${GAME_SERVER_TOKEN}`;
            const isValid = authHeader && authHeader.length === expectedToken.length &&
                crypto.timingSafeEqual(Buffer.from(authHeader), Buffer.from(expectedToken));
            if (!isValid) {
                console.log(`⛔ Unauthorized access attempt.`);
                return res.status(401).json({ error: "Unauthorized: Invalid or missing token." });
            }
        }

        // 2. Validate Payload
        const {
            playerInput,
            currentStoryContext,
            possibleReturnPoints,

            existingBackgrounds,
            existingBgms,

            characterState // NEW
        } = req.body;

        if (!playerInput) {
            return res.status(400).json({ error: "Missing playerInput" });
        }

        if (!API_KEY) {
            throw new Error("Server configuration error: API_KEY is missing.");
        }

        console.log(`📝 Processing Branch: "${playerInput}" (Node: ${currentStoryContext.id})`);

        // --- Semantic Cache Check ---
        const history = await loadHistory();
        const nodeHistory = history[currentStoryContext.id] || [];

        // Extract Names for semantic check
        const characterNames = characters ? characters.map(c => c.name) : [];

        // Check if there's a match

            playerInput,
            nodeHistory, // Pass full history objects (input + state)
            currentStoryContext,
            characterNames,
            characterState,
            API_KEY

            }
        }

        if (!success || !aiRes) {
            throw lastError || new Error("All AI models failed to generate content.");
        }

        const aiData = await aiRes.json();

        // Extract text from Gemini-compatible response
        const candidate = aiData.candidates?.[0];

        // Check for truncation or safety blocks
        if (candidate?.finishReason === 'MAX_TOKENS') {
            console.warn("⚠️ AI response was truncated due to MAX_TOKENS! Consider increasing maxOutputTokens or simplifying the prompt.");
        } else if (candidate?.finishReason === 'SAFETY') {
            console.warn("⚠️ AI response was blocked by safety filters.");
        }

        // Handle multi-part responses (filtering out thinking process)
        const parts = candidate?.content?.parts || [];
        const rawText = parts
            .filter(p => p.thought !== true)
            .map(p => p.text)
            .join('');

        if (!rawText) {
            throw new Error("Empty response from AI provider" + (candidate?.finishReason ? ` (finishReason: ${candidate.finishReason})` : ''));
        }

        let jsonResult;
        try {
            const cleanedText = cleanJsonString(rawText);
            jsonResult = JSON.parse(cleanedText);
        } catch (parseError) {
            console.error("❌ JSON Parse Failed. Raw Text Length:", rawText.length);
            // Log the start of the text to see if it's thinking
            console.error("❌ Raw Snippet:", rawText.substring(0, 500));
            throw new Error(`AI returned invalid JSON: ${parseError.message}`);
        }

        // --- OUTPUT PROCESSING (Enrich Response with Stored URLs) ---
        if (jsonResult.newBackgrounds && Array.isArray(jsonResult.newBackgrounds)) {
            bgLibDirty = false;

            jsonResult.newBackgrounds = jsonResult.newBackgrounds.map(newBg => {
                // Try exact ID match first
                let storedBg = bgLibrary[newBg.id];

                // Fallback: partial name matching (e.g. name contains [刹那] matches 九条 刹那)
                if (!storedBg && newBg.name) {
                    const matchEntry = Object.values(bgLibrary).find(lib => {
                        if (!lib.name) return false;
                        // Check if either name contains the other (partial match)
                        return lib.name.includes(newBg.name) || newBg.name.includes(lib.name);
                    });
                    if (matchEntry) {
                        console.log(`🔗 Name-matched background: "${newBg.name}" ↔ "${matchEntry.name}" (${matchEntry.id})`);
                        storedBg = matchEntry;
                    }
                }

                // Case A: Known background with URL
                if (storedBg && storedBg.url) {
                    console.log(`🔗 Attaching existing URL to AI background: ${newBg.id} (matched: ${storedBg.id})`);
                    return { ...newBg, url: storedBg.url };
                }

                // Case B: Truly new background (or known but no URL) -> Store definition
                if (!storedBg) {
                    bgLibrary[newBg.id] = newBg;
                    bgLibDirty = true;
                } else if (newBg.description !== storedBg.description) {
                    // Update description if AI refined it
                    bgLibrary[storedBg.id] = { ...storedBg, description: newBg.description };
                    bgLibDirty = true;
                }

                return newBg;
            });

            if (bgLibDirty) await saveBgLibrary(bgLibrary);
        }

        // 5. Save to File (Persistence)
        const createHash = (str) => crypto.createHash('sha256').update(str).digest('hex').substring(0, 12);

        const shortHash = createHash(playerInput + JSON.stringify(characterState || {}));
        const rawTitle = jsonResult.chapterTitle || playerInput.slice(0, 10);
        const safeTitle = rawTitle.replace(/[^a-z0-9\u4e00-\u9fa5]/gi, '_').slice(0, 20);

        const safeStoryId = currentStoryContext.id ? currentStoryContext.id.replace(/[^a-z0-9_\-\u4e00-\u9fa5]/gi, '_') : 'unknown_story';
        const filename = `${safeStoryId}_${safeTitle}_${shortHash}.json`;
        const filePath = path.join(DATA_DIR, filename);

        const fileData = {
            meta: {
                timestamp: Date.now(),
                date: new Date().toISOString(),
                playerInput,
                aiGuidance,
                processTimeMs: Date.now() - startTime
            },
            requestContext: {
                currentStory: currentStoryContext.name,
                currentStoryId: currentStoryContext.id
            },
            result: jsonResult
        };

        await fs.writeFile(filePath, JSON.stringify(fileData, null, 2));
        console.log(`💾 Branch saved to: data/${filename}`);

        // --- UPDATE HISTORY INDEX (with lock to prevent concurrent write corruption) ---
        await withHistoryLock(async () => {
            const freshHistory = await loadHistory(); // Re-read to avoid stale data
            if (!freshHistory[currentStoryContext.id]) {
                freshHistory[currentStoryContext.id] = [];
            }
            freshHistory[currentStoryContext.id].push({
                input: playerInput,
                state: characterState, // Store state for future semantic matching
                filename: filename,
                timestamp: Date.now(),
                count: 1 // Init Tracking
            });
            await saveHistory(freshHistory);
        });

        // 6. Return Response (include branch meta for voting)
        res.json({
            ...jsonResult,
            _branchMeta: {
                filename: filename,
                storyId: currentStoryContext.id,
                likes: 0,
                dislikes: 0
            }
        });

    } catch (e) {
        console.error("❌ Server Error:", e);
        res.status(500).json({ error: e.message || "Internal Server Error" });
    }
};

// Returns a random cached branch for a given storyId
app.get('/api/random-branch', async (req, res) => {
    try {
        // Auth check (same as generate)
        const authHeader = req.headers.authorization;
        if (GAME_SERVER_TOKEN) {
            const expectedToken = `Bearer ${GAME_SERVER_TOKEN}`;
            const isValid = authHeader && authHeader.length === expectedToken.length &&
                crypto.timingSafeEqual(Buffer.from(authHeader), Buffer.from(expectedToken));
            if (!isValid) {
                return res.status(401).json({ error: "Unauthorized" });
            }
        }

        const storyId = req.query.storyId;
        if (!storyId) {
            return res.status(400).json({ error: "Missing storyId query parameter" });
        }

        const history = await loadHistory();
        const entries = history[storyId];
        if (!entries || entries.length === 0) {
            return res.status(404).json({ error: "No cached branches for this story node", found: false });
        }

        // Pick random entry
        const randomEntry = entries[Math.floor(Math.random() * entries.length)];
        const filePath = path.join(DATA_DIR, randomEntry.filename);

        try {
            const fileContent = await fs.readFile(filePath, 'utf-8');
            const fileData = JSON.parse(fileContent);

            console.log(`🎲 Random branch served: "${randomEntry.input}" from ${randomEntry.filename}`);

            res.json({
                found: true,
                playerInput: randomEntry.input,
                result: fileData.result,
                totalBranches: entries.length,
                _branchMeta: {
                    filename: randomEntry.filename,
                    storyId,
                    likes: randomEntry.votes?.likes || 0,
                    dislikes: randomEntry.votes?.dislikes || 0
                }
            });
        } catch (fileErr) {
            console.error(`❌ Failed to read branch file: ${randomEntry.filename}`, fileErr);
            return res.status(500).json({ error: "Branch file not found or corrupted" });
        }
    } catch (e) {
        console.error("❌ Random Branch Error:", e);
        res.status(500).json({ error: e.message || "Internal Server Error" });
    }
});

// --- NEW: Vote Endpoint (Like/Dislike AI Branches) ---
app.post('/api/vote', async (req, res) => {
    try {
        // Auth check
        const authHeader = req.headers.authorization;
        if (GAME_SERVER_TOKEN) {
            const expectedToken = `Bearer ${GAME_SERVER_TOKEN}`;
            const isValid = authHeader && authHeader.length === expectedToken.length &&
                crypto.timingSafeEqual(Buffer.from(authHeader), Buffer.from(expectedToken));
            if (!isValid) {
                return res.status(401).json({ error: "Unauthorized" });
            }
        }

        const { storyId, filename, vote, previousVote } = req.body;
        if (!storyId || !filename || !['like', 'dislike'].includes(vote)) {
            return res.status(400).json({ error: 'Missing or invalid parameters. Required: storyId, filename, vote (like/dislike)' });
        }

        await withHistoryLock(async () => {
            const history = await loadHistory();
            const nodeHistory = history[storyId];
            if (!nodeHistory) {
                return res.status(404).json({ error: 'Story node not found in history' });
            }

            const entryIndex = nodeHistory.findIndex(e => e.filename === filename);
            if (entryIndex === -1) {
                return res.status(404).json({ error: 'Branch entry not found' });
            }

            const entry = nodeHistory[entryIndex];

            // Initialize vote counts if not present
            if (!entry.votes) entry.votes = { likes: 0, dislikes: 0 };

            // Undo previous vote if switching (client tells us what to undo)
            if (previousVote === 'like') entry.votes.likes = Math.max(0, entry.votes.likes - 1);
            if (previousVote === 'dislike') entry.votes.dislikes = Math.max(0, entry.votes.dislikes - 1);

            // Apply new vote
            if (vote === 'like') entry.votes.likes++;
            if (vote === 'dislike') entry.votes.dislikes++;

            console.log(`🗳️ Vote: ${vote} on "${entry.input}" (${filename}) | 👍${entry.votes.likes} 👎${entry.votes.dislikes}`);

            // --- AUTO-REMOVAL CHECK: dislikes > likes + 10 ---
            if (entry.votes.dislikes > entry.votes.likes + 10) {
                console.log(`🗑️ Auto-removing low-quality branch: "${entry.input}" (👍${entry.votes.likes} 👎${entry.votes.dislikes})`);

                // Remove data file
                try {
                    const filePath = path.join(DATA_DIR, entry.filename);
                    await fs.unlink(filePath);
                    console.log(`🗑️ Deleted file: ${entry.filename}`);
                } catch (err) {
                    console.warn(`⚠️ Could not delete file ${entry.filename}:`, err.message);
                }

                // Remove from history
                nodeHistory.splice(entryIndex, 1);
                await saveHistory(history);

                return res.json({
                    likes: entry.votes.likes,
                    dislikes: entry.votes.dislikes,
                    removed: true,
                    message: '该剧情因差评过多已被自动移除'
                });
            }

            await saveHistory(history);

            res.json({
                likes: entry.votes.likes,
                dislikes: entry.votes.dislikes,
                removed: false
            });
        });
    } catch (e) {
        console.error("❌ Vote Error:", e);
        res.status(500).json({ error: e.message || "Internal Server Error" });
    }
});
